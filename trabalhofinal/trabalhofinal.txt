# ---------------------------
# HashTable: separate chaining
# ---------------------------
class HashTable:
    """
    HashTable com separate chaining. Aceita chaves int ou str.
    Implementado para evitar uso de dict como armazenamento direto.
    """
    def __init__(self, m=131071):  # tamanho inicial (primo aproximado)
        self.m = int(m)
        self.table = [[] for _ in range(self.m)]
        self._size = 0

    def _hash(self, key):
        # usa hash builtin e mapeia para [0, m-1]
        h = hash(key)
        # garantir inteiro positivo
        return (h & 0x7FFFFFFFFFFFFFFF) % self.m

    def put(self, key, value):
        h = self._hash(key)
        bucket = self.table[h]
        for i, (k, v) in enumerate(bucket):
            if k == key:
                bucket[i] = (k, value)
                return
        bucket.append((key, value))
        self._size += 1

    def get(self, key):
        h = self._hash(key)
        bucket = self.table[h]
        for k, v in bucket:
            if k == key:
                return v
        return None

    def contains(self, key):
        return self.get(key) is not None

    def append_to_listvalue(self, key, value):
        """
        Se a chave não existir, cria lista; se existir, adiciona ao final.
        Útil para índices invertidos.
        """
        h = self._hash(key)
        bucket = self.table[h]
        for i, (k, v) in enumerate(bucket):
            if k == key:
                v.append(value)
                return
        # não achou: inserir nova lista
        bucket.append((key, [value]))
        self._size += 1

    def items(self):
        for bucket in self.table:
            for k, v in bucket:
                yield (k, v)

    def keys(self):
        for bucket in self.table:
            for k, _ in bucket:
                yield k

    def __len__(self):
        return self._size

# ---------------------------
# MovieRecord
# ---------------------------
class MovieRecord:
    __slots__ = ('movieId', 'title', 'genres', 'rating_sum', 'rating_count', 'avg')

    def __init__(self, movieId, title, genres):
        self.movieId = int(movieId)
        self.title = title
        # armazenar gêneros como lista (minúsculos)
        if isinstance(genres, str):
            self.genres = [g for g in genres.split('|') if g != '(no genres listed)'] if genres else []
        elif isinstance(genres, list):
            self.genres = genres
        else:
            self.genres = []
        self.genres = [g.strip().lower() for g in self.genres if g.strip() != '']
        self.rating_sum = 0.0
        self.rating_count = 0
        self.avg = 0.0

    def add_rating(self, r):
        self.rating_sum += float(r)
        self.rating_count += 1

    def finalize(self):
        if self.rating_count:
            self.avg = self.rating_sum / self.rating_count
        else:
            self.avg = 0.0

# ---------------------------
# Trie (for prefix search of titles)
# ---------------------------
class TrieNode:
    __slots__ = ('children', 'movie_ids', 'is_end')
    def __init__(self):
        # usar dict para filhos por caractere (aceitável e eficiente)
        self.children = {}
        self.movie_ids = []  # lista de movieIds que terminam exatamente aqui (ou que estão associados)
        self.is_end = False

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, title, movieId, store_on_path=False):
        """
        Insere título no trie. Se store_on_path=True, armazena o movieId em todos os nós do path —
        isso acelera buscas por prefixo mas consome mais memória.
        """
        node = self.root
        t = title.lower()
        for ch in t:
            if ch not in node.children:
                node.children[ch] = TrieNode()
            node = node.children[ch]
            if store_on_path:
                node.movie_ids.append(movieId)
        node.is_end = True
        node.movie_ids.append(movieId)

    def starts_with(self, prefix):
        node = self.root
        p = prefix.lower()
        for ch in p:
            if ch not in node.children:
                return []
            node = node.children[ch]
        # coletar todos movie_ids na subtrie
        res = []
        stack = [node]
        while stack:
            n = stack.pop()
            if n.movie_ids:
                res.extend(n.movie_ids)
            for child in n.children.values():
                stack.append(child)
        return res

# ---------------------------
# Indexes: UserIndex, TagIndex
# ---------------------------
class UserIndex:
    def __init__(self, table_size=262139):
        self.ht = HashTable(m=table_size)

    def add(self, userId, movieId, rating):
        key = int(userId)
        rec = self.ht.get(key)
        if rec is None:
            self.ht.put(key, [(int(movieId), float(rating))])
        else:
            rec.append((int(movieId), float(rating)))

    def get(self, userId):
        rec = self.ht.get(int(userId))
        if rec is None:
            return []
        return rec

class TagIndex:
    def __init__(self, table_size=131071):
        self.ht = HashTable(m=table_size)

    def add(self, tag, movieId):
        t = tag.strip().lower()
        if t == '':
            return
        self.ht.append_to_listvalue(t, int(movieId))

    def get(self, tag):
        return self.ht.get(tag.strip().lower()) or []

# ---------------------------
# Construção das estruturas
# ---------------------------
def build_structures(movies_csv, ratings_csv, tags_csv, chunksize=200_000, show_progress=True):
    """
    Constrói:
      - movies_ht: HashTable movieId -> MovieRecord
      - trie: Trie de títulos
      - user_index: UserIndex
      - tag_index: TagIndex
    Leitura por chunks nas tabelas grandes (ratings, tags) usando pandas se disponível.
    """
    t0 = time.time()
    movies_ht = HashTable(m=400009)
    trie = Trie()
    user_index = UserIndex()
    tag_index = TagIndex()

    # 1) Ler movies.csv (normalmente pequeno: id,title,genres)
    if show_progress: print("Lendo movies...")
    with open(movies_csv, newline='', encoding='utf-8') as f:
        reader = csv.DictReader(f)
        for row in reader:
            try:
                mid = int(row['movieId'])
                title = row.get('title', '')
                genres = row.get('genres', '')
            except Exception:
                # tentar índices alternativos
                vals = list(row.values())
                if len(vals) >= 3:
                    mid = int(vals[0])
                    title = vals[1]
                    genres = vals[2]
                else:
                    continue
            mr = MovieRecord(mid, title, genres)
            movies_ht.put(mid, mr)
            # inserir no trie; não armazenamos em cada nó do caminho para economizar memória
            trie.insert(title, mid, store_on_path=False)

    # 2) Ler ratings.csv em chunks
    if show_progress: print("Lendo ratings (por chunks) e populando contagens...")
    if PANDAS_AVAILABLE:
        # pandas lida bem com grandes arquivos em chunks
        for chunk in pd.read_csv(ratings_csv, usecols=['userId','movieId','rating'], chunksize=chunksize):
            for row in chunk.itertuples(index=False):
                try:
                    uid = int(row[0])
                    mid = int(row[1])
                    r = float(row[2])
                except Exception:
                    continue
                # atualizar movie record se existir
                mr = movies_ht.get(mid)
                if mr is not None:
                    mr.add_rating(r)
                # armazenar rating do usuário (mesmo que filme desconhecido)
                user_index.add(uid, mid, r)
    else:
        # fallback: leitura streaming com csv.DictReader linha a linha
        with open(ratings_csv, newline='', encoding='utf-8') as f:
            reader = csv.DictReader(f)
            for i,row in enumerate(reader, start=1):
                try:
                    uid = int(row['userId'])
                    mid = int(row['movieId'])
                    r = float(row['rating'])
                except Exception:
                    vals = list(row.values())
                    if len(vals) >= 3:
                        uid = int(vals[0]); mid = int(vals[1]); r = float(vals[2])
                    else:
                        continue
                mr = movies_ht.get(mid)
                if mr is not None:
                    mr.add_rating(r)
                user_index.add(uid, mid, r)
                if show_progress and i % 500000 == 0:
                    print(f"  {i} ratings lidos...")

    # 3) Ler tags.csv (pode ser grande) e popular tag_index
    if tags_csv:
        if show_progress: print("Lendo tags...")
        if PANDAS_AVAILABLE:
            # Alguns arquivos tags.csv têm colunas (userId,movieId,tag,timestamp); pegamos movieId e tag
            for chunk in pd.read_csv(tags_csv, usecols=['movieId','tag'], chunksize=chunksize):
                for row in chunk.itertuples(index=False):
                    try:
                        mid = int(row[0])
                        tag = str(row[1])
                    except Exception:
                        continue
                    tag_index.add(tag, mid)
        else:
            with open(tags_csv, newline='', encoding='utf-8') as f:
                reader = csv.DictReader(f)
                for i,row in enumerate(reader, start=1):
                    try:
                        mid = int(row['movieId'])
                        tag = row['tag']
                    except Exception:
                        vals = list(row.values())
                        if len(vals) >= 3:
                            mid = int(vals[1]); tag = vals[2]
                        else:
                            continue
                    tag_index.add(tag, mid)
                    if show_progress and i % 200000 == 0:
                        print(f"  {i} tags lidos...")

    # 4) calcular médias finais
    if show_progress: print("Calculando médias por filme...")
    for mid, mr in movies_ht.items():
        mr.finalize()

    t1 = time.time()
    if show_progress:
        print(f"Build concluído em {t1 - t0:.2f}s.")
        print(f"Filmes carregados: {len(list(movies_ht.keys()))}")
        print(f"Usuários indexados (estimado): {len(user_index.ht)}")
        print(f"Tags distintas (estimado): {len(tag_index.ht)}")
    return movies_ht, trie, user_index, tag_index

# ---------------------------
# Funções de consulta e formatação
# ---------------------------
def format_movie_row(mr):
    genres = ", ".join(mr.genres) if mr.genres else "-"
    return f"{mr.movieId:<8}\t{mr.title[:60]:60}\t{genres:20}\t{mr.avg:9.6f}\t{mr.rating_count:6d}"

def query_prefix(trie, movies_ht, prefix, limit=200):
    ids = trie.starts_with(prefix)
    # remover duplicatas e manter ordem única
    unique = {}
    for mid in ids:
        unique[mid] = True
    mids = list(unique.keys())
    # buscar registros
    recs = []
    for mid in mids:
        mr = movies_ht.get(mid)
        if mr:
            recs.append(mr)
    # ordenar por média desc, em caso de empate por contagem desc
    recs.sort(key=lambda x: (-x.avg, -x.rating_count, x.title))
    total = len(recs)
    if total == 0:
        print("Nenhum filme encontrado para o prefixo informado.")
        return
    to_show = recs[:limit]
    print(f"Mostrando {len(to_show)} de {total} resultados para prefixo '{prefix}':")
    print("movieId  \ttitle                                                         \tgenres               \t   avg   \t count")
    for mr in to_show:
        print(format_movie_row(mr))
    if total > limit:
        print(f"... (omitidos {total - limit} resultados)")

def query_user(user_index, movies_ht, userId, limit=50):
    recs = user_index.get(userId)
    if not recs:
        print(f"Nenhum rating encontrado para o usuário {userId}.")
        return
    # cada rec: (movieId, rating_by_user)
    full = []
    for (mid, ur) in recs:
        mr = movies_ht.get(mid)
        if mr:
            full.append((mid, ur, mr.avg, mr.rating_count, mr.title))
        else:
            full.append((mid, ur, 0.0, 0, "<movie not in movies.csv>"))
    # ordenar: rating do usuário desc, média global desc
    full.sort(key=lambda x: (-x[1], -x[2], -x[3]))
    print(f"Ratings do usuário {userId} (mostrando até {limit}):")
    print("movieId  \ttitle                                                         \tuser_rating\tglobal_avg\tcount")
    for item in full[:limit]:
        mid, ur, gavg, cnt, title = item
        print(f"{mid:<8}\t{title[:60]:60}\t{ur:11.6f}\t{gavg:11.6f}\t{cnt:6d}")

def query_top_genre(movies_ht, N, genre, mincount=0):
    g = genre.strip().lower()
    results = []
    for mid, mr in movies_ht.items():
        if mr is None:
            continue
        if g in mr.genres and mr.rating_count >= mincount:
            results.append(mr)
    results.sort(key=lambda x: (-x.avg, -x.rating_count, x.title))
    print(f"Top {N} filmes do gênero '{genre}' (mincount={mincount}):")
    print("movieId  \ttitle                                                         \tgenres               \t   avg   \t count")
    for mr in results[:N]:
        print(format_movie_row(mr))
    if len(results) == 0:
        print("Nenhum filme encontrado para esse gênero (verifique ortografia).")

def query_tags(tag_index, movies_ht, tags_list, limit=200):
    tags_list = [t.strip().lower() for t in tags_list if t.strip() != ""]
    if not tags_list:
        print("Forneça pelo menos uma tag.")
        return
    # pegar listas para cada tag
    lists = []
    for t in tags_list:
        lst = tag_index.get(t)
        if not lst:
            print(f"Nenhum filme encontrado para a tag '{t}'. Interseção vazia.")
            return
        lists.append(lst)
    # interseção: começar pelo menor
    lists.sort(key=lambda x: len(x))
    s = set(lists[0])
    for lst in lists[1:]:
        s.intersection_update(lst)
        if not s:
            print("Interseção vazia — nenhum filme tem todas as tags informadas.")
            return
    # buscar MovieRecords
    recs = []
    for mid in s:
        mr = movies_ht.get(mid)
        if mr:
            recs.append(mr)
    recs.sort(key=lambda x: (-x.avg, -x.rating_count, x.title))
    total = len(recs)
    print(f"Filmes com tags {tags_list}: {total} resultados. Mostrando até {limit}:")
    print("movieId  \ttitle                                                         \tgenres               \t   avg   \t count")
    for mr in recs[:limit]:
        print(format_movie_row(mr))
    if total > limit:
        print(f"... (omitidos {total - limit} resultados)")

def print_stats(movies_ht, user_index, tag_index):
    # estatísticas simples
    total_movies = len(list(movies_ht.keys()))
    total_users = len(user_index.ht)
    total_tags = len(tag_index.ht)
    avg_ratings_per_movie = 0.0
    cnt = 0
    for mid, mr in movies_ht.items():
        avg_ratings_per_movie += mr.rating_count
        cnt += 1
    if cnt:
        avg_ratings_per_movie /= cnt
    print("ESTATÍSTICAS:")
    print(f"  Filmes carregados: {total_movies}")
    print(f"  Usuários indexados (estima): {total_users}")
    print(f"  Tags distintas (estima): {total_tags}")
    print(f"  Média de ratings por filme: {avg_ratings_per_movie:.2f}")

# ---------------------------
# Console interativo
# ---------------------------
def repl(movies_ht, trie, user_index, tag_index):
    banner = """
Console interativo pronto.
Comandos:
  prefixo <string>               - busca títulos que começam com <string>
  user <userId>                  - mostra ratings do usuário (ordenado)
  top <N> <genero> [mincount]    - mostra top N filmes do gênero (opcional mincount)
  tags <tag1> <tag2> ...         - interseção por tags (separadas por espaço)
  stats                          - estatísticas básicas
  help                           - este help
  exit | quit                    - sair
Observações:
  - Strings (prefixos, gêneros, tags) são case-insensitive.
  - As médias são mostradas com 6 casas decimais.
"""
    print(banner)
    while True:
        try:
            s = input("cmd> ").strip()
        except EOFError:
            print()
            break
        if not s:
            continue
        parts = s.split()
        cmd = parts[0].lower()
        args = parts[1:]
        try:
            if cmd in ('exit', 'quit'):
                print("Saindo.")
                break
            elif cmd == 'help':
                print(banner)
            elif cmd == 'prefixo':
                if not args:
                    print("Uso: prefixo <string>")
                    continue
                prefix = " ".join(args)
                query_prefix(trie, movies_ht, prefix)
            elif cmd == 'user':
                if not args:
                    print("Uso: user <userId>")
                    continue
                uid = int(args[0])
                query_user(user_index, movies_ht, uid)
            elif cmd == 'top':
                if len(args) < 2:
                    print("Uso: top <N> <genero> [mincount]")
                    continue
                N = int(args[0])
                genre = args[1]
                mincount = int(args[2]) if len(args) >= 3 else 0
                query_top_genre(movies_ht, N, genre, mincount=mincount)
            elif cmd == 'tags':
                if not args:
                    print("Uso: tags <tag1> <tag2> ...")
                    continue
                query_tags(tag_index, movies_ht, args)
            elif cmd == 'stats':
                print_stats(movies_ht, user_index, tag_index)
            else:
                print("Comando não reconhecido. Digite 'help' para ver os comandos.")
        except Exception as e:
            print("Erro durante execução do comando:", e)

# ---------------------------
# Main / argparse
# ---------------------------
def main():
    p = argparse.ArgumentParser(description="Programa de busca/consulta sobre filmes (construção offline e console).")
    p.add_argument("--movies", required=True, help="Arquivo movies.csv")
    p.add_argument("--ratings", required=True, help="Arquivo ratings.csv")
    p.add_argument("--tags", required=False, default=None, help="Arquivo tags.csv (opcional)")
    p.add_argument("--chunksize", required=False, type=int, default=200_000, help="Chunksize para pandas (se disponível)")
    p.add_argument("--no-progress", action='store_true', help="Não mostrar progresso")
    args = p.parse_args()

    show_prog = not args.no_progress
    print("Construindo estruturas (pode demorar dependendo do dataset).")
    movies_ht, trie, user_index, tag_index = build_structures(
        args.movies, args.ratings, args.tags, chunksize=args.chunksize, show_progress=show_prog
    )
    repl(movies_ht, trie, user_index, tag_index)

if __name__ == "__main__":
    main()
